/***            USER            ***/
GET /api/user/{email} -> zwrócenie klienta z danym emailem
       TO FRONT-END
            email - string - id
            name - string
            surname - string
            password - string
            wish_list - List<String> (Book_Id)

GET /api/users -> zwrócenie listy klientow
        TO FRONT-END
            lista wszystkich userów - List<User>

POST /api/user -> tworzy klienta
        TO AND FROM FRONT-END
            email - string - id
            name - string
            surname - string
            password - string
            wish_list - List<String> (Book_Id)

PUT /api/user/{email} -> updateuje klienta
        TO AND FROM FRONT-END
            email - string - id
            name - string
            surname - string
            password - string
            wish_list - List<String> (Book_Id)

DELETE /api/user/{email} -> usuwa klienta pod danym emailem
        TO FRONT-END
            true / false

GET /api/user/getWishList?user={email} -> zwraca wishList klienta
        TO FRONT-END
            wysh_list - List<String> - lista identyfikatorów książek

        TO FRONT-END:

            avaliable: <bool>

        TO AND FROM FRONT-END:

            publishHouse: <String>
            condiiton: <String>



/***        ACCOUNT     ***/

GET /api/account?email={email}&password={password} -> zwraca dane konta z podanym loginem i hasłem
        TO FRONT-END
            email - string - id
            name - string
            surname - string
            password - string
            wish_list - List<String> (Book_Id) (w przypadku User)
            role - string [24] {librarian, admin} (w przypadku Employee)
